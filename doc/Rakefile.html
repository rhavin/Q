    # load specifiation from gemspec and create # a task for every extension file in # executables. spec = Gem::Specification.load(‘Q.gemspec’) spec.executables.each do |f|

    Rake::ExtensionTask.new('Q', spec) do |ext|
  # extension file is *.so, so we need

  ext.name    = f.gsub(%r\.so$/,'')
  ext.tmp_dir = 'tmp'
  ext.lib_dir = 'bin'
