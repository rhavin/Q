    This is an extented sprintf-function that allowes usage of _type substituted by bold type and _S substituted as code-notifier

    # File lib/Q/console.rb, line 56
def Console.txFormat(str, arrVals)
  myStr = str.dup()
  myStr.gsub!(%r%(\d*)\$_S/, txCode('%\1$s'))
  myStr.gsub!(%r%(\d*)\$_(.)/, txBold('%\1$\2'))
  sprintf(myStr, *arrVals)
end

    txSanitize!(str) click to toggle source

    reinterpret ansi-color ‘default’ as last set color in given string this is simply because most terminals dont implement a color-stack

    # File lib/Q/console.rb, line 65
def Console.txSanitize!(str)
  color = [9]
  str.gsub!(%r\e\[3([^m]+)m/) do
    if ($1 == '9')
      color.pop()
      c = color.last
    else
      color.push($1)
      c = $1
    end
    "\e[3#{c}m"
  end
end
